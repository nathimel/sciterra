 27def batch_cospsi_matrix(embeddings: np.ndarray) -> np.ndarray:
 28    """Batch-process a pairwise cosine similarity matrix between embeddings.
 29
 30    In order to avoid memory errors (e.g. bus error, segfaults) resulting from too large arrays, we batch process the construction of the cospsi_matrix.
 31
 32    Args:
 33        embeddings: a 1D numpy array of embeddings
 34
 35    Returns:
 36        cosine_similarities: a 2D numpy array representing the pairwise cosine similarity between each embedding
 37    """
 38    batch_size = min(1000, len(embeddings))  # Define a batch size
 39
 40    cosine_similarities = None
 41    print(
 42        f"computing cosine similarity for {len(embeddings)} embeddings with batch size {batch_size}."
 43    )
 44    for i in tqdm(range(0, len(embeddings), batch_size)):
 45        # Process batches to compute cosine similarity
 46        batch = embeddings[i : i + batch_size]
 47        if cosine_similarities is None:
 48            cosine_similarities = cosine_similarity(batch, embeddings)
 49        else:
 50            cosine_similarities = np.vstack(
 51                (cosine_similarities, cosine_similarity(batch, embeddings))
 52            )
 53
 54    return cosine_similarities

 86    def bibtex_to_atlas(self, bibtex_fp: str, *args, **kwargs) -> Atlas:
 87        """Convert a bibtex file to an atlas, by parsing each entry for an identifier, and querying an API for publications using `self.librarian`.
 88
 89        NOTE: the identifiers in the corresponding atlas will be API-specific ids; there is no relationship between the parsed id used to query papers (e.g. 'DOI:XYZ' in the case of SemanticScholar) and the resulting identifier associated with the resulting Publication object, (a paperId, a bibcode, etc.) Therefore, the purpose of using the `bibtex_to_atlas` method is primarily for initializing literature exploration in a human-readable way. If you want to obtain as many publications as identifiers supplied, you need to use `get_publications`.
 90
 91        Args:
 92            bibtex_fp: the filepath where the bibtex file is saved.
 93
 94            args and kwargs are passed to `get_publications`.
 95        """
 96        verbose = get_verbose(kwargs)
 97
 98        with open(bibtex_fp, "r") as f:
 99            bib_database = bibtexparser.load(f)
100
101        # Retrieve the identifier from each bibtex entry
102        identifiers = [
103            self.librarian.bibtex_entry_identifier(entry)
104            for entry in bib_database.entries
105        ]
106        identifiers = [id for id in identifiers if id is not None]
107        if len(identifiers) < len(bib_database.entries) and verbose:
108            warnings.warn(
109                f"Only obtained {len(identifiers)} publications out of {len(bib_database.entries)} total due to missing identifiers."
110            )
111
112        # Query
113        results = self.librarian.get_publications(identifiers, *args, **kwargs)
114        # Validate
115        publications = [
116            result
117            for result in results
118            if (
119                result is not None
120                and result.publication_date is not None
121                and result.abstract is not None
122                # identifier will never be none
123            )
124        ]
125        if len(publications) < len(identifiers) and verbose:
126            warnings.warn(
127                f"Only obtained {len(publications)} publications out of {len(identifiers)} total due to querying-related errors or missing abstracts."
128            )
129
130        # Construct atlas
131        atl = Atlas(publications)
132        return atl

138    def project(self, atl: Atlas, **kwargs) -> Atlas:
139        """Update an atlas with its projection, i.e. the document embeddings for all publications using `self.vectorizer`, removing publications with no abstracts.
140
141        Args:
142            atl: the Atlas containing publications to project to document embeddings
143
144        Returns:
145            the updated atlas containing all nonempty-abstract-containing publications and their projection
146        """
147        verbose = get_verbose(kwargs)
148
149        # Only project publications that have abstracts and publication dates
150        atl_filtered = self.filter_by_attributes(atl, **kwargs)
151        num_empty = len(atl) - len(atl_filtered)
152        if num_empty and verbose:
153            warnings.warn(
154                f"{num_empty} publications were filtered due to missing crucial data. There are now {len(atl_filtered.bad_ids)} total ids that will be excluded in the future."
155            )
156
157        # Project
158        embeddings = None
159        # get only embeddings for publications not already projected in atlas
160        previously_embedded_ids = []
161        if atl_filtered.projection is not None:
162            previously_embedded_ids = atl_filtered.projection.identifier_to_index
163        embed_ids = [
164            id for id in atl_filtered.publications if id not in previously_embedded_ids
165        ]
166
167        if embed_ids:
168            if verbose:
169                if atl_filtered.projection is not None:
170                    warnings.warn(
171                        f"Found {len(atl_filtered.publications) - len(atl_filtered.projection)} publications not contained in Atlas projection."
172                    )
173                warnings.warn(f"Embedding {len(embed_ids)} total documents.")
174            # Embed documents
175            result = self.vectorizer.embed_documents(
176                [atl_filtered[id].abstract for id in embed_ids]
177            )
178            embeddings = result["embeddings"]
179            successful_indices = result["indices"]
180
181            if len(successful_indices) < len(embed_ids) and verbose:
182                warnings.warn(f"Failed to get embeddings for all {len(embed_ids)} publications; only {len(embeddings)} will be added to the Atlas.")
183
184            successful_ids = [id for i, id in enumerate(embed_ids) if i in successful_indices]
185
186            # create new projection
187            projection = Projection(
188                identifier_to_index={
189                    identifier: idx for idx, identifier in enumerate(successful_ids)
190                },
191                index_to_identifier=tuple(successful_ids),
192                embeddings=embeddings,
193            )
194
195        if not embed_ids or embed_ids is None and verbose:
196            warnings.warn(f"Obtained no new publication embeddings.")
197            projection = get_empty_projection()
198
199        # merge existing projection with new projection
200        merged_projection = merge(atl_filtered.projection, projection)
201
202        # prepare to overwrite atlas with publications corresponding to updated (merged) projection
203        embedded_publications = {
204            id: pub
205            for id, pub in atl_filtered.publications.items()
206            if id in merged_projection.identifier_to_index
207        }
208        assert not set(atl_filtered.ids()) - set(embedded_publications.keys())
209
210        # Overwrite atlas data
211        atl_filtered.publications = embedded_publications
212        atl_filtered.projection = merged_projection
213        return atl_filtered

219    def expand(
220        self,
221        atl: Atlas,
222        *args,
223        center: str = None,
224        n_pubs_max: int = 4000,
225        n_sources_max: int = None,
226        record_pubs_per_update: bool = False,
227        **kwargs,
228    ) -> Atlas:
229        """Expand an atlas by retrieving a list of publications resulting from traversal of the citation network.
230
231        Args:
232            atl: the atlas containing the region to expand
233
234            center: (if given) center the search on this publication, preferentially searching related publications.
235
236            n_pubs_max: maximum number of publications allowed in the expansion.
237
238            n_sources_max: maximum number of publications (already in the atlas) to draw references and citations from.
239
240            record_pubs_per_update: whether to track all the publications that exist in the resulting atlas to `self.pubs_per_update`. This should only be set to `True` when you need to later filter by degree of convergence of the atlas.
241
242        Returns:
243            atl_expanded: the expanded atlas
244        """
245        existing_keys = set(atl.ids())
246        expand_keys = existing_keys
247        if center is not None:
248            # If atlas is initial
249            if atl.projection is None:
250                atl = self.project(atl)
251
252            if len(atl.projection):
253                # build cosine similarity matrix, of shape (1, num_pubs)
254                cospsi_matrix = cosine_similarity(
255                    atl.projection.identifiers_to_embeddings([center]),
256                    atl.projection.embeddings,
257                )
258                # get most similar keys from center, including center itself
259                sort_inds = np.argsort(cospsi_matrix)[::-1][
260                    0
261                ]  # argsort orders from least to greatest similarity, so reverse
262                expand_keys = atl.projection.indices_to_identifiers(sort_inds)
263
264        if n_sources_max is not None:
265            expand_keys = expand_keys[:n_sources_max]
266
267        # Get identifiers for the expansion
268        # For each publication corresponding to an id in `expand_keys`, collect the ids corresponding to the publication's references and citations.
269        ids = set()
270        for key in expand_keys:
271            ids_i = set(atl[key].references + atl[key].citations)
272            # Prune for obvious overlap, and for ids that have previously failed
273            ids.update(ids_i - existing_keys - atl.bad_ids)
274            # Break when the search is centered and we're maxed out
275            if len(ids) > n_pubs_max and center is not None:
276                break
277        ids = list(ids)
278
279        if not ids:
280            print("Overly-restrictive search, no ids to retrive.")
281
282        # Sample to account for max number of publications we want to retrieve
283        if len(ids) > n_pubs_max:
284            ids = np.random.choice(ids, n_pubs_max, replace=False)
285
286        print(f"Expansion will include {len(ids)} new publications.")
287
288        # Retrieve publications from ids using a librarian
289        new_publications = self.librarian.get_publications(ids, *args, **kwargs)
290
291        # New atlas
292        atl_exp = Atlas(new_publications)
293
294        # Update the new atlas
295        atl_exp.publications.update(atl.publications)
296        atl_exp.bad_ids = atl.bad_ids
297        atl_exp.projection = (
298            atl.projection
299        )  # new projection will be updated in `project`
300
301        # Record the new list of publications
302        if record_pubs_per_update:
303            self.pubs_per_update.append(list(atl_exp.ids()))
304
305        return atl_exp

311    def filter_by_attributes(
312        self,
313        atl: Atlas,
314        attributes: list = [
315            "abstract",
316            "publication_date",
317        ],
318        record_pubs_per_update=False,
319        **kwargs,
320    ) -> Atlas:
321        """Update an atlas by dropping publications (and corresponding data in projection) when certain fields are empty.
322
323        Args:
324            atl: the Atlas containing publications to filter
325
326            attributes: the list of attributes to filter publications from the atlas IF any of items are None for a publication. For example, if attributes = ["abstract"], then all publications `pub` such that `pub.abstract is None` is True will be removed from the atlas, along with the corresponding data in the projection.
327
328            record_pubs_per_update: whether to track all the publications that exist in the resulting atlas to `self.pubs_per_update`. This should only be set to `True` when you need to later filter by degree of convergence of the atlas. This is an important parameter because `self.filter` is called in `self.project`, which typically is called after `self.expand`, where we pass in the same parameter.
329
330        Returns:
331            the filtered atlas
332        """
333        # Filter publications
334        invalid_pubs = {
335            id: pub
336            for id, pub in atl.publications.items()
337            if (pub is None or any([getattr(pub, attr) is None for attr in attributes]))
338        }
339        # Do not update if unnecessary
340        if not len(invalid_pubs):
341            return atl
342
343        atl_filtered = self.filter_by_ids(atl, drop_ids=invalid_pubs.keys())
344
345        # Record only the publications in the history that weren't filtered out
346        if record_pubs_per_update:
347            self.pubs_per_update[-1] = atl_filtered.ids()
348
349        return atl_filtered

351    def filter_by_ids(
352        self,
353        atl: Atlas,
354        keep_ids: list[str] = None,
355        drop_ids: list[str] = None,
356    ) -> Atlas:
357        """Update an atlas by dropping publications (and corresponding data in projection).
358
359        Args:
360            atl: the Atlas containing publications to filter
361
362            keep_ids: the list of publication ids to NOT filter; all other publications in `atl` not matching one of these ids will be removed.
363
364            drop_ids: the list of publications to filter; all publications in `atl` matching one of these ids will be removed.
365        """
366
367        if all(x is not None for x in [keep_ids, drop_ids]):
368            raise ValueError(
369                "You must pass exactly one of `keep_ids` or `drop_ids`, but both had a value that was not `None`."
370            )
371        if keep_ids is not None:
372            filter_ids = set([id for id in atl.ids() if id not in keep_ids])
373        elif drop_ids is not None:
374            filter_ids = set(drop_ids)
375        else:
376            raise ValueError(
377                "You must pass exactly one of `keep_ids` or `drop_ids`, but both had value `None`."
378            )
379
380        # Keep track of the bad identifiers to skip them in future expansions
381        new_bad_ids = atl.bad_ids.union(filter_ids)
382
383        # Filter embeddings, ids from projection
384        if atl.projection is None:
385            new_projection = None
386        else:
387            filter_indices = set()
388            idx_to_id_new = []
389            # From indexing
390            for idx, id in enumerate(atl.projection.index_to_identifier):
391                if id in filter_ids:
392                    filter_indices.add(idx)
393                else:
394                    idx_to_id_new.append(id)
395            # From embeddings
396            embeddings = np.array(
397                [
398                    embedding
399                    for idx, embedding in enumerate(atl.projection.embeddings)
400                    if idx not in filter_indices
401                ]
402            )
403            # From identifier to index map
404            id_to_idx_new = {id: idx for idx, id in enumerate(idx_to_id_new)}
405            # Construct new, filtered projection
406            new_projection = Projection(
407                identifier_to_index=id_to_idx_new,
408                index_to_identifier=idx_to_id_new,
409                embeddings=embeddings,
410            )
411
412        # Keep only filtered publications
413        new_publications = [
414            pub for id, pub in atl.publications.items() if id not in filter_ids
415        ]
416
417        # Construct new atlas
418        atl_filtered = Atlas(new_publications, new_projection)
419        atl_filtered.bad_ids = new_bad_ids
420
421        return atl_filtered

452    def record_update_history(
453        self,
454        pubs: list[str] = None,
455        pubs_per_update: list[list[str]] = None,
456    ) -> None:
457        """Record when publications were added, by updating atl.update_history.
458
459        atl.update_history is a np.array of ints representing when publications were added. A value of -2 indicates no record of being added.
460
461        Args:
462            pubs: a list of str ids corresponding to publications at the final update in the update history. By default `None`, and `self.pubs_per_update[-1]` will be used.
463
464            pubs_per_update: a list of which publications existed at which iteration, with the index of the overall list corresponding to the iteration the publication was added. By default `None`, and `self.pubs_per_update` will be used.
465
466        Updates:
467            `self.update_history`: an np.array of ints representing when publications were added. A value of -2 indicates no record of being added.
468
469        Returns:
470            `None`
471        """
472        if pubs is None:
473            pubs = self.pubs_per_update[-1]
474
475        if pubs_per_update is None:
476            pubs_per_update = self.pubs_per_update
477
478        # Loop backwards
479        i_max = len(pubs_per_update) - 1
480        update_history = np.full(len(pubs), -2)
481        for i, pubs_i in enumerate(pubs_per_update[::-1]):
482            is_in = np.isin(pubs, pubs_i)
483            update_history[is_in] = i_max - i
484        self.update_history = update_history

490    def converged_kernel_size(self, atl: Atlas) -> np.ndarray:
491        """Calculate the largest size of the kernel that's converged (at differing levels of convergence) for each publication in a sample at each update.
492
493        Args:
494            atl: Atlas containing publications; for each publication we compute the largest converged kernel size at each update
495
496        Returns:
497            kernel_size: an array of ints of shape `(num_pubs, max_update)` representing the kernel size for converged kernels.
498                - The first column indicates the largest kernel size that hasn't changed since the beginning,
499                - The second column indicates the largest kernel size that hasn't changed since the first update,
500                - etc. for the nth column.
501        """
502
503        if self.update_history is None:
504            raise ValueError(
505                "update_history is None; make sure you have called record_update_history()!"
506            )
507
508        if -2 in self.update_history:
509            raise ValueError(
510                "Incomplete update history as indicated by entries with values of -2."
511            )
512
513        publications = np.array(atl.ids())
514
515        # 1. Loop over each publication
516        cospsi_kernel = []
517        for pub in tqdm(publications):
518            # 2. Identify the similarity with the other publications relative to this publication, and sort accordingly.
519            cospsi = cosine_similarity(
520                atl.projection.identifiers_to_embeddings([pub]),
521                atl.projection.embeddings,
522            ).flatten()  # shape `(num_pubs,)`
523            sort_inds = np.argsort(cospsi)[::-1]  # shape `(num_pubs,)`
524
525            # 3. Identify the expansion iteration at which those publications were added to the atlas (`sorted_history`).
526            sorted_history = self.update_history[sort_inds]  # shape `(num_pubs,)`
527
528            # 4. Identify the latest iteration at which any publication was added to the atlas; this can be less than the total iterations.
529            last_update = self.update_history.max()
530
531            # 5. Loop through each iteration until `last_update`, and identify which publications were added at or before that iteration.
532            result_2 = [
533                # 6. Compute how many publications out we can go and still only contain publications added at or before that iteration.
534                # Use `argmin` to get the first instance of False
535                # Finally, subtract 1: we want the first index before False.
536                np.argmin(sorted_history <= update) - 1
537                for update in range(last_update)
538            ]  # shape `(num_pubs, last_update)`
539
540            cospsi_kernel.append(result_2)
541
542        return np.array(cospsi_kernel)

548    def measure_topography(
549        self,
550        atl: Atlas,
551        ids: list[str] = None,
552        metrics: list[str] = ["density"],
553        min_prior_pubs: int = 2,
554        kernel_size=16,
555        **kwargs,
556    ):
557        """Measure topographic properties of all publications relative to prior
558        publications.
559
560        Args:
561
562            atl: the Atlas to measure
563
564            publication_indices: an np.ndarray of ints representing the indices of publications in the Atlas projection to measure
565
566            metrics: A list of strings representing the metrics to use. Options are...
567                constant_asymmetry: The asymmetry of a publication $p_i$ w.r.t the entire atlas $\\{ p_j \\forall j \\in \\{1, ..., k\\} \\} where $k$ is the length of the atlas
568
569                    $| \\sum_{j}^{k-1}( p_i - p_j ) |$
570
571                kernel_constant_asymmetry: The asymmetry of a publication w.r.t. its kernel, { p_j for all j in {1, ..., k} } where k is `kernel_size`, i.e. the k nearest neighbors.
572
573                density: the density of a publication's surrounding area, estimated by a heuristic inspired by mass / volume = k publications divided by the minimum arc length enclosing the furthest publication.
574
575                    $\\frac{ k }{ smoothing\\_length(k) }$
576
577                smoothing_length: The distance (in radians) to the farthest publication in the kernel, i.e. the kth nearest neighbor.
578
579            min_prior_pubs: The minimum number of publications prior to the target publication for which to calculate the metric.
580
581            kernel_size: the number of publications surrounding the publication for which to compute the topography metric, i.e. k nearest neighbors for k=kernel_size.
582
583        Returns:
584            estimates: an np.ndarray of shape `(len(publication_indices), len(metrics))` representing the estimated topography metric values for each publication.
585        """
586        verbose = get_verbose(kwargs)
587
588        # By default calculate for all publications
589        if ids is None:
590            ids = atl.ids()
591        else:
592            ids = list(ids)
593
594        if not ids:
595            raise Exception("No publications to measure topography of.")
596
597        # Get publication dates, for filtering
598        dates = np.array([atl[identifier].publication_date for identifier in ids])
599
600        # Get pairwise cosine similarities for ids
601        embeddings = atl.projection.identifiers_to_embeddings(ids)
602        cospsi_matrix = batch_cospsi_matrix(embeddings)
603
604        # From here on, use embedding indices instead of identifiers
605        # our embeddings are already in the correct order, so just use them
606        publication_indices = np.arange(len(embeddings))
607        # publication_indices = atl.projection.identifiers_to_indices(ids)
608
609        print(f"Computing {metrics} for {len(ids)} publications.")
610        estimates = []
611        # for idx in tqdm(publication_indices):
612        for idx, identifier in enumerate(ids):
613            # Get the date of publication
614            # identifier = atl.projection.index_to_identifier[idx]
615            date = atl[identifier].publication_date
616
617            # Identify prior publications
618            is_prior = dates < date
619            if is_prior.sum() < min_prior_pubs:
620                estimates.append([np.nan for _ in metrics])
621                continue
622
623            # Choose valid publications
624            is_other = publication_indices != idx
625            is_valid = is_prior & is_other
626            valid_indices = publication_indices[is_valid]
627
628            kwargs = {
629                "idx": idx,
630                "cospsi_matrix": cospsi_matrix,
631                "valid_indices": valid_indices,
632                "publication_indices": publication_indices,
633                "embeddings": embeddings,
634                "kernel_size": kernel_size,
635            }
636
637            def call_metric(
638                metric: str,
639                **kwargs,

491    def converged_kernel_size(self, atl: Atlas) -> np.ndarray:
492        """Calculate the largest size of the kernel that's converged (at differing levels of convergence) for each publication in a sample at each update.
493
494        Args:
495            atl: Atlas containing publications; for each publication we compute the largest converged kernel size at each update
496
497        Returns:
498            kernel_size: an array of ints of shape `(num_pubs, max_update)` representing the kernel size for converged kernels.
499                - The first column indicates the largest kernel size that hasn't changed since the beginning,
500                - The second column indicates the largest kernel size that hasn't changed since the first update,
501                - etc. for the nth column.
502        """
503
504        if self.update_history is None:
505            raise ValueError(
506                "update_history is None; make sure you have called record_update_history()!"
507            )
508
509        if -2 in self.update_history:
510            raise ValueError(
511                "Incomplete update history as indicated by entries with values of -2."
512            )
513
514        publications = np.array(atl.ids())
515
516        # 1. Loop over each publication
517        cospsi_kernel = []
518        for pub in tqdm(publications):
519            # 2. Identify the similarity with the other publications relative to this publication, and sort accordingly.
520            cospsi = cosine_similarity(
521                atl.projection.identifiers_to_embeddings([pub]),
522                atl.projection.embeddings,
523            ).flatten()  # shape `(num_pubs,)`
524            sort_inds = np.argsort(cospsi)[::-1]  # shape `(num_pubs,)`
525
526            # 3. Identify the expansion iteration at which those publications were added to the atlas (`sorted_history`).
527            sorted_history = self.update_history[sort_inds]  # shape `(num_pubs,)`
528
529            # 4. Identify the latest iteration at which any publication was added to the atlas; this can be less than the total iterations.
530            last_update = self.update_history.max()
531
532            # 5. Loop through each iteration until `last_update`, and identify which publications were added at or before that iteration.
533            result_2 = [
534                # 6. Compute how many publications out we can go and still only contain publications added at or before that iteration.
535                # Use `argmin` to get the first instance of False
536                # Finally, subtract 1: we want the first index before False.
537                np.argmin(sorted_history <= update) - 1
538                for update in range(last_update)
539            ]  # shape `(num_pubs, last_update)`
540
541            cospsi_kernel.append(result_2)
542
543        return np.array(cospsi_kernel)

528

530            last_update = self.update_history.max()
531
532            # 5. Loop through each iteration until `last_update`, and identify which publications were added at or before that iteration.
533            result_2 = [

535                # Use `argmin` to get the first instance of False
536                # Finally, subtract 1: we want the first index before False.
537                np.argmin(sorted_history <= update) - 1
538                for update in range(last_update)
539            ]  # shape `(num_pubs, last_update)`
